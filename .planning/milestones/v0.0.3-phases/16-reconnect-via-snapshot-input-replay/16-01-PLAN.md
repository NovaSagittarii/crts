    ```typescript
    replayInputLog(entries: InputLogEntry[]): void {
      if (!this.rtsRoom || entries.length === 0) {
        return;
      }

      const sorted = [...entries].sort(
        (a, b) => a.tick - b.tick || a.sequence - b.sequence,
      );

      for (const entry of sorted) {
        if (entry.kind === 'build') {
          this.applyQueuedBuild(entry.payload as BuildQueuedPayload);
        } else if (entry.kind === 'destroy') {
          this.applyQueuedDestroy(entry.payload as DestroyQueuedPayload);
        }
      }
    }
    ```

    Note: The `as BuildQueuedPayload` / `as DestroyQueuedPayload` casts are necessary because `InputLogEntry.payload` is typed as `unknown`. This is safe because the server stores the actual typed payload objects, and Socket.IO JSON serialization preserves the shape. Add eslint-disable comment for the unsafe argument if needed (the test file already has these disables at the top).
