    ```typescript
    export interface RewardConfig {
      weights: {
        terminal: number;       // default: 1.0
        economy_delta: number;  // default: 0.1
        core_damage: number;    // default: 0.5
      };
      annealEpisodes: number;   // default: 10000
    }

    export const DEFAULT_REWARD_CONFIG: RewardConfig = {
      weights: { terminal: 1.0, economy_delta: 0.1, core_damage: 0.5 },
      annealEpisodes: 10000,
    };

    export interface RewardStateSnapshot {
      resources: number;
      income: number;
      coreHp: number;
      enemyCoreHp: number;
    }
    ```

    Export `computeReward` pure function:
    ```typescript
    export function computeReward(
      prev: RewardStateSnapshot,
      curr: RewardStateSnapshot,
      terminated: boolean,
      truncated: boolean,
      isWinner: boolean | null,  // null = draw or not finished
      config: RewardConfig,
      episodeNumber: number,
    ): number
    ```

    **Implementation (per D-09, D-10, D-11, D-12):**

    1. Terminal reward:
       - If `terminated || truncated`: `isWinner === true` => `+1.0 * config.weights.terminal`, `isWinner === false` => `-1.0 * config.weights.terminal`, `isWinner === null` => `0.0`
    2. Shaped rewards:
       - `shapedWeight = Math.max(0, 1.0 - episodeNumber / config.annealEpisodes)`
       - If `shapedWeight > 0`:
         - `economyDelta = ((curr.resources + curr.income) - (prev.resources + prev.income)) / 100`
         - `coreDamage = (prev.enemyCoreHp - curr.enemyCoreHp) / 500`
         - Add `shapedWeight * (config.weights.economy_delta * economyDelta + config.weights.core_damage * coreDamage)`
    3. Return total reward

    This is a pure function -- no internal state, no side effects. Annealing state (episodeNumber) is passed in by the caller (BotEnvironment or training loop).

    8. **custom weights** -- double economy_delta weight, verify reward doubles for economy component
    9. **no change tick** -- identical prev and curr, not terminal, reward = 0.0
